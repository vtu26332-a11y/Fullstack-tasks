package com.jobportal.repository; import com.jobportal.entity.Job; import com.jobportal.entity.User; import org.springframework.data.domain.Page; import org.springframework.data.domain.Pageable; import org.springframework.data.domain.Sort; import org.springframework.data.jpa.repository.JpaRepository; import org.springframework.data.jpa.repository.Query; import org.springframework.stereotype.Repository; import java.math.BigDecimal; import java.util.List; @Repository public interface JobRepository extends JpaRepository { List findByEmployer(User employer); List findByLocationContainingIgnoreCase(String location, Sort sort); List findByCategoryContainingIgnoreCase(String category, Sort sort); List findByExperienceContainingIgnoreCase(String experience, Sort sort); List findBySkillsContainingIgnoreCase(String skills, Sort sort); @Query(""" SELECT j FROM Job j WHERE (:category IS NULL OR LOWER(j.category) LIKE LOWER(CONCAT('%', :category, '%'))) AND (:location IS NULL OR LOWER(j.location) LIKE LOWER(CONCAT('%', :location, '%'))) AND (:experience IS NULL OR LOWER(j.experience) LIKE LOWER(CONCAT('%', :experience, '%'))) AND (:skills IS NULL OR LOWER(j.skills) LIKE LOWER(CONCAT('%', :skills, '%'))) AND (:minSalary IS NULL OR j.salary >= :minSalary) AND (:maxSalary IS NULL OR j.salary <= :maxSalary) """) Page search( String category, String location, String experience, String skills, BigDecimal minSalary, BigDecimal maxSalary, Pageable pageable ); }
